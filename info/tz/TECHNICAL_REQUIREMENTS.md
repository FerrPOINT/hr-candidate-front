  static memoize<T extends (...args: any[]) => any>(
    fn: T,
    deps: any[]
  ): T {

    return ((...args: any[]) => {
      if (lastArgs && this.shallowEqual(lastArgs, args)) {
        return lastResult;
      }
      
      lastArgs = args;
      lastResult = fn(...args);
      return lastResult;
    }) as T;

  // –õ–µ–Ω–∏–≤–∞—è –∑–∞–≥—Ä—É–∑–∫–∞ –¥–∞–Ω–Ω—ã—Ö
  static lazyLoad<T>(

    options: {
      cacheTime?: number;
      staleTime?: number;
    } = {}
  ) {
    let data: T | null = null;
    let loading = false;
    let error: Error | null = null;
    let lastFetch = 0;

    return {
      async load() {
        const now = Date.now();
        
        // –ü—Ä–æ–≤–µ—Ä—è–µ–º –∫—ç—à
        if (data && (now - lastFetch) < cacheTime) {
          return data;

          error = null;

          loading = false;
